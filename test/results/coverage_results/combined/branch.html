val branch : (word, bool) -> bool
function branch(op, branch_condition) = {
    incr_cycles(2);

    let next_PC = reg_PC;
    let word = calc_word(mode(rel), op);
    let new_PC = next_PC + signed(word);

    if branch_condition then {
        reg_PC = new_PC;

        if (new_PC[15..8] == next_PC[15..8]) then
            incr_cycles(1)
        else
            incr_cycles(2);
    };

    true;
}
